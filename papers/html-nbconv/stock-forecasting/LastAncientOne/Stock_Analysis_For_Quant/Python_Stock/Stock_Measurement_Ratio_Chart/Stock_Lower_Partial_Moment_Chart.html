def lpm(stock_returns):
    threshold=0.0
    order=1
    threshold_array = np.empty(len(stock_returns))
    threshold_array.fill(threshold)
    diff = threshold_array - stock_returns
    diff = diff.clip()
    return np.sum(diff ** order) / len(stock_returns)
