def sortino_ratio(symbol):
    numer = pow((1 + symbol.mean()), 252) - 1
    annual_volatility = symbol.std() * np.sqrt(252)
    denom = annual_volatility

    if denom > 0.0:
         sortino_ratio = numer / denom
    else:
        print('none')
    return sortino_ratio
