n = 20
df['MF_Multiplier'] = (2*df['Adj Close'] - df['Low'] - df['High'])/(df['High']-df['Low'])
df['MF_Volume'] = df['MF_Multiplier']*df['Volume']
df['CMF'] = df['MF_Volume'].rolling(n).sum()/df['Volume'].rolling(n).sum()
df = df.drop(['MF_Multiplier','MF_Volume'],axis=1)
