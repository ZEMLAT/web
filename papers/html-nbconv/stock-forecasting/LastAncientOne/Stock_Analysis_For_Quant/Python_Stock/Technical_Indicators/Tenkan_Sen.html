# Tenkan-sen (Conversion Line): (9-Period High + 9-Period Low)/2))
Nine_Period_High = df['High'].rolling(window=9).max()
Nine_Period_Low = df['Low'].rolling(window=9).min()
df['Tenkan_Sen'] = (Nine_Period_High + Nine_Period_Low)/2
