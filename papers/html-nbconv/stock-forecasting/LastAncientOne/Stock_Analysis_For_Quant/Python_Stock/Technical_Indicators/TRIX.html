n = 15 # Number of days
df['SS_EMA'] = df['Adj Close'].rolling(n).mean()
df['DS_EMA'] = df['SS_EMA'].rolling(n).mean()
df['TS_EMA'] = df['DS_EMA'].rolling(n).mean()
df['TRIX'] = 100*(df['TS_EMA'] - df['TS_EMA'].shift())/df['TS_EMA'].shift()
df = df.drop(['SS_EMA','DS_EMA','TS_EMA'],axis=1)
