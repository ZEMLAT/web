def SMMA(price, n, m=3):
    SMMA = np.array([np.nan] * len(price))
    SMMA[n - 2] = price[:n - 1].mean()
    for i in range(n - 1, len(price)):
        SMMA[i] = (SMMA [i - 1] * (n - 2) + 2 * price[i]) / n
    return SMMA
