n = 14 # number of days
s = 3 # smoothing
df['High_Highest'] = df['Adj Close'].rolling(n).max()
df['Low_Lowest'] = df['Adj Close'].rolling(n).min()
df['Fast_%K'] = ((df['Adj Close'] - df['Low_Lowest']) / (df['High_Highest'] - df['Low_Lowest'])) * 100
df['Fast_%D'] = df['Fast_%K'].rolling(s).mean()
