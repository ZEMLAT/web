n = 14
df['Lowest Low'] = df['Low'].rolling(n).min()
df['Highest High'] = df['High'].rolling(n).max()
df['William%R'] = -100*(df['Highest High'] - df['Adj Close'])/(df['Highest High'] - df['Lowest Low'])
df = df.drop(['Lowest Low','Highest High'],axis=1)
