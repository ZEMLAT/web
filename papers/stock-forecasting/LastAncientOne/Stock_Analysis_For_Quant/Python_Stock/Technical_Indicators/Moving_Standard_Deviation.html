n = 20
df['MA20'] = df['Adj Close'][0:n].mean()
df['MA_STD'] = df['Adj Close'] - df['MA20']
df['MA_STD2'] = df['MA_STD']**2
df['MA_STD'] = df['MA_STD2']/df['MA_STD2'].rolling(n).sum()
