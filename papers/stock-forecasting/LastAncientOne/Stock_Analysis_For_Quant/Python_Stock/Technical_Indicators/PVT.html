df['Momentum_1D'] = (df['Adj Close'] - df['Adj Close'].shift(1)).fillna(0)
df["PVT"] = (df['Momentum_1D']/ df['Adj Close'].shift(1))*df['Volume']
df["PVT"] = df["PVT"] - df["PVT"].shift(1)
df["PVT"] = df["PVT"].fillna(0)
