from matplotlib import dates as mdates
import datetime as dt

dfc = df.copy()
dfc['QStick'] = (dfc['Adj Close'] - dfc['Open']).rolling(10).mean() 
dfc['VolumePositive'] = dfc['Open'] < dfc['Adj Close']
dfc = dfc.dropna()
dfc = dfc.reset_index()
dfc['Date'] = mdates.date2num(dfc['Date'].astype(dt.date))
