n = 14
smin = df['Low'].rolling(n).min()
smax = df['High'].rolling(n).max()
df['stoch_k'] = 100 * (df['Adj Close'] - smin) / (smax - smin)
d_n = 3
df['stoch_d'] = df['stoch_k'].rolling(d_n).mean()
